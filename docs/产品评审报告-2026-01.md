│    服务层 (services/ + actions/)     │  ← 业务逻辑

**代码示例**（优秀错误处理）:
```typescript
async executeMacro(macroId: number, triggerType: string): Promise<void> {
  const startTime = Date.now();
  try {
    // 1. 加载宏配置
    const macro = await this.dbService.getMacro(macroId);
    if (!macro || !macro.enabled) {
      Logger.warn(TAG, 'Macro not found or disabled');
      return;
    }

    // 2. 创建执行上下文
    const context = await ExecutionContext.create(macro);

    // 3. 评估条件
    const conditionsPassed = await this.conditionEvaluator.evaluate(context);
    if (!conditionsPassed) {
      Logger.info(TAG, 'Conditions not met, skip execution');
      return;
    }

    // 4. 顺序执行动作
    for (const action of macro.actions) {
      await this.actionExecutor.execute(action, context);
    }

    // 5. 记录成功日志
    await this.dbService.addExecutionLog({
      macroId, triggerType, status: 'success',
      duration: Date.now() - startTime
    });

    this.notificationService.show('宏执行成功', macro.name);
  } catch (error) {
    Logger.error(TAG, 'Macro execution failed', error as Error);
    await this.dbService.addExecutionLog({
      macroId, triggerType, status: 'failed',
      errorMessage: (error as Error).message
    });
    this.notificationService.show('宏执行失败', (error as Error).message);
  }
}
```

### 6.1 阻塞性问题（P0）🔴

#### **问题1: 后台触发能力未实现**

**现状**: `TriggerBackgroundAbility` 仅声明，未实现

**影响**:
- 定时触发器无法在后台运行
- 网络触发器无法在后台监听
- **核心功能不可用**

**修复方案**:
```typescript
// entry/src/main/ets/entryability/TriggerBackgroundAbility.ts
import ServiceExtensionAbility from '@ohos.app.ability.ServiceExtensionAbility';
import workScheduler from '@ohos.resourceschedule.workScheduler';
import MacroEngine from '../services/MacroEngine';
import Logger from '../utils/Logger';

const TAG = 'TriggerBackgroundAbility';

export default class TriggerBackgroundAbility extends ServiceExtensionAbility {
  onCreate(want: Want): void {
    Logger.info(TAG, 'Service created');
  }

  onWorkStart(workInfo: workScheduler.WorkInfo): void {
    Logger.info(TAG, 'Work triggered', JSON.stringify(workInfo));

    const macroId = workInfo.parameters?.macroId as number;
    const triggerType = workInfo.parameters?.triggerType as string;

    if (!macroId) {
      Logger.error(TAG, 'Invalid macroId');
      return;
    }

    // 执行宏
    MacroEngine.getInstance()
      .executeMacro(macroId, triggerType)
      .then(() => {
        Logger.info(TAG, 'Macro executed successfully');
      })
      .catch(error => {
        Logger.error(TAG, 'Macro execution failed', error as Error);
      });
  }

  onWorkStop(workInfo: workScheduler.WorkInfo): void {
    Logger.info(TAG, 'Work stopped', JSON.stringify(workInfo));
  }

  onDestroy(): void {
    Logger.info(TAG, 'Service destroyed');
  }
}
```

**工作量**: 2 小时

**优先级**: 🔴 **立即修复**
